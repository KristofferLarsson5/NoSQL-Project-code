    + _id: ObjectId

    + _id: ObjectId
    + name: String
    + contactDetails: String
}

class Customer {
    + _id: ObjectId
    + name: String
    + contactDetails: String
    + favorites: Recipe[]
}

class Recipe {
    + _id: ObjectId
    + name: String
    + id: String
    + instructions: String
    + origin: String
    + dietaryRestrictions: String[]
    + comments: Comment[]
    + ratings: Rating[]
    + createdBy: ObjectId
}

class Comment {
    + body: String
    + date: Date
    + customer: ObjectId
}
